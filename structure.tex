% Определение цветов для подсветки
\definecolor{codegreen}{rgb}{0,0.6,0}
\definecolor{codegray}{rgb}{0.5,0.5,0.5}
\definecolor{codepurple}{rgb}{0.58,0,0.82}
\definecolor{backcolour}{rgb}{0.95,0.95,0.92}

\lstset{
    backgroundcolor=\color{backcolour},   % цвет фона
    commentstyle=\color{codegreen},       % цвет комментариев
    keywordstyle=\color{blue}\bfseries,   % цвет ключевых слов
    numberstyle=\tiny\color{codegray},    % стиль номеров строк
    stringstyle=\color{codepurple},       % цвет строк
    basicstyle=\small\ttfamily,           % моноширинный шрифт
    breakatwhitespace=false,         
    breaklines=true,                      % автоматический перенос строк
    captionpos=b,                         % заголовок снизу
    keepspaces=true,                      % сохранять пробелы и отступы
    numbers=left,                         % нумерация строк слева
    numbersep=5pt,                  
    showspaces=false,                
    showstringspaces=false,
    showtabs=false,                  
    tabsize=4,                            % размер табуляции
    frame=single,                         % рамка вокруг кода
    % Дополнительные ключевые слова для C++ и Eigen
    morekeywords={VectorXd, MatrixXd, FullPivLU, vector, pair,
                 iota, emplace_back, auto}
}
